  import {mapGetters} from 'vuex'
  import uploadFile from '../../components/uploadFile'
  export default {
    name: 'channelManage',
    components: {
      uploadFile
    },
    data: function () {
      return {
        state:'',
        channelList: [],
        inBranchName: '',
        showImport: false,
        visible: false,
        pageNum: 1,
        pageSize: 10,
        total: 0,
        dialogVisible: false,
        form:{
          id: '',
          branchName: '',
          numPersonnel: '',
          contactName: '',
          contactTelephone: '',
          examineState: '',
        },
        applyStatusTxt:{
          '1': '通过',
          '2': '驳回',
          '3': '待审核',
        },
        auditingId: '',
      }
    },
    computed:{
      ...mapGetters(['user'])
    },
    created() {
      this.getChannelList()
    },
    methods: {
      command(command) {
        if (command.type === 1) {
          this.edit(command.data)
        } else if (command.type === 2) {
          this.auditing(command.data)
        } else {
          this.retPad(command.data)
        }
      },
      assign1(){
        let params = {
          id: this.auditingId,
          examineState:'4'
        }
        this.$http.post('/c621060branchinfo/check',params).then(res=>{
          this.$message({
            message: res.msg,
            type: 'success'
          })
          this.dialogVisible = false
          this.getChannelList()
        },error=>{
          console.log(error)
        })
      },
      assign2(){
        let params = {
          id: this.auditingId,
          examineState:'3'
        }
        this.$http.post('/c621060branchinfo/check',params).then(res=>{
          this.$message({
            message: res.msg,
            type: 'success'
          })
          this.dialogVisible = false
          this.getChannelList()
        },error=>{
          console.log(error)
        })
      },
      url1(){
        let params = {
          upfile:this.upfile
        }
        this.$http.post('/c621060branchinfo/list', params).then(res => {
          this.channelList = res.ls
          this.total = res.page.resultCount
        }, error => {
          console.log(error)
        })
      },
      clickSearch(){
        this.pageNum = 1
        this.getChannelList()
      },
      getChannelList() {
        let params = {
          examineState: this.state,
          branchName:this.inBranchName,
          pageCurrent:this.pageNum,
          pageLimit:this.pageSize
        }
        this.$http.post('/c621060branchinfo/list', params).then(res => {
          this.channelList = res.data
          this.total = res.page.resultCount
          console.log(res.data)
        }, error => {
          console.log(error)
        })
      },
      gotoPage(path) {
        this.$router.push(path)
      },
      cellStyle() {
        return 'text-align:left'
      },
      view(item){
        this.$router.push({path: '/channel/channelView', query: item})
      },
      edit(item) {
        this.$router.push({path: '/channel/edit', query: item})
      },
      toon() {
        let params = {
          id: this.branchName,
          examineState: this.examineState,
        }
        this.$http.post('/c621060branchinfo/check', params).then(res => {
          this.$message({
            message: res.msg,
            type: 'success'
          })
          this.getChannelList()
        }, error => {
          console.log(error)
        })
      },
      tableHeaderStyle(){
        return 'background-color: #e3ecfc;font-size: 14px;font-weight: 600;color: #2d4667;text-align: center;'

        return 'text-align: center;font-size: 12px;color: #707070;'
      },
      addEvenRowClass({row,rowIndex}){
        if((rowIndex%2) !== 0){
          return 'evenRow'

        })
      },
